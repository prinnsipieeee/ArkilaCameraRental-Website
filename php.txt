// Function to handle updates for each table
function updateTable($conn, $table, $data, $whereColumn) {
    unset($data['update_customer'], $data['update_booking'], $data['update_product'], $data['update_sale'], $data['update_users']);

    $setValues = [];
    foreach ($data as $column => $value) {
        $setValues[] = "$column = '" . $conn->real_escape_string($value) . "'"; 
    }
    $setValuesStr = implode(", ", $setValues);
    $whereValue = $conn->real_escape_string($data[$whereColumn]);
    $updateQuery = "UPDATE $table SET $setValuesStr WHERE $whereColumn = '$whereValue'";

    if ($conn->query($updateQuery) === TRUE) {
        return ['status' => 'success', 'message' => "Record updated successfully in $table"];
    } else {
        error_log("Error in query: " . $updateQuery); // Log query for debugging
        return ['status' => 'error', 'message' => "Error updating $table: " . $conn->error];
    }

    $updateQuery = "UPDATE $table SET $setValuesStr WHERE $whereColumn = '$whereValue'";
    error_log("Executing query: $updateQuery"); // Log the query
}

    unset($data['table']);        // Remove the table name flag

    // Use the appropriate where column depending on the table
    $whereColumn = isset($data['users_id']) ? 'users_id' : (isset($data['customer_id']) ? 'customer_id' : (isset($data['booking_id']) ? 'booking_id' : ((isset($data['product_id']) ? 'product_id' : 'sale_id'))));
    
    $updateResult = updateTable($conn, $table, $data, $whereColumn);
    
    header('Content-Type: application/json'); // Specify JSON content type
    echo json_encode($updateResult); // Return JSON response
    exit; // Stop further script execution
}

        function toggleTable(tableId) {
            var tables = document.querySelectorAll('.table-section');

        function updateRecord(event, table) {
            event.preventDefault();  // Prevent form submission

            var form = event.target;
            var formData = new FormData(form);
            formData.append('ajax_update', true);
            formData.append('table', table);

            fetch('dashboard.php', {
                method: 'POST',
                body: formData
            })
            .then(response => response.json())
            .then(data => {
                // Handle the success or error message
                var messageDiv = form.querySelector('.message');
                if (data.status === 'success') {
                    messageDiv.textContent = data.message;
                    messageDiv.style.color = 'green';
                } else {
                    messageDiv.textContent = data.message;
                    messageDiv.style.color = 'red';
                }
            })
            .catch(error => {
                console.error('Error:', error);
            });
        }
    </script>

     <script>
    function updateRecord(form, tableName) {
        // Log form and tableName for debugging
        console.log("Updating record in table:", tableName);

        // Convert form data to JSON
        const formData = new FormData(form);
        formData.append('ajax_update', true);
        formData.append('table', tableName);

        fetch('dashboard.php', {
            method: 'POST',
            body: formData,
        })
        .then(response => response.json())
        .then(data => {
            console.log("Response data:", data); // Log response for debugging
            if (data.status === 'success') {
                alert(data.message);  // Show success message
                location.reload();    // Reload the page to show updated data
            } else {
                alert(data.message);  // Show error message
            }
        })
        .catch(error => {
            console.error('Error:', error);
            alert('An error occurred while updating the record.');
        });

        return false;  // Prevent default form submission
        }
